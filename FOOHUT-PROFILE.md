# FooHut Profile System
## Comprehensive Feature Specification v1.0

---

## 1. Executive Summary

The FooHut Profile System is the social identity layer that powers community, collaboration, and professional networking across the platform. It transforms FooHut from a docs/dev tool into a living ecosystem where developers, security professionals, and teams connect, collaborate, and build together.

**Core Philosophy:** *Your profile is your developer passport â€” showcase your work, connect with peers, and build your reputation.*

```sql
-- =====================================================
-- PROFILE SYSTEM TABLES
-- =====================================================

-- Users table (core)
CREATE TABLE users (
  id TEXT PRIMARY KEY DEFAULT (uuid()),
  username TEXT UNIQUE NOT NULL,
  email TEXT UNIQUE NOT NULL,
  display_name TEXT NOT NULL,
  avatar_url TEXT,
  cover_url TEXT,
  bio TEXT,
  tagline TEXT,
  pronouns TEXT,
  location TEXT,
  timezone TEXT,
  
  -- Professional
  profession TEXT,
  company TEXT,
  company_role TEXT,
  company_verified BOOLEAN DEFAULT FALSE,
  company_verified_at DATETIME,
  years_experience INTEGER,
  work_status TEXT, -- 'open', 'freelance', 'exploring', 'not_looking', 'hiring'
  hourly_rate INTEGER,
  
  -- Visibility
  profile_visibility TEXT DEFAULT 'public', -- 'public', 'friends', 'private'
  
  -- Metadata
  created_at DATETIME DEFAULT CURRENT_TIMESTAMP,
  updated_at DATETIME DEFAULT CURRENT_TIMESTAMP,
  last_seen_at DATETIME,
  onboarding_completed BOOLEAN DEFAULT FALSE
);

-- User social links
CREATE TABLE user_social_links (
  id TEXT PRIMARY KEY DEFAULT (uuid()),
  user_id TEXT NOT NULL REFERENCES users(id) ON DELETE CASCADE,
  platform TEXT NOT NULL, -- 'website', 'github', 'linkedin', etc.
  url TEXT NOT NULL,
  verified BOOLEAN DEFAULT FALSE,
  display_order INTEGER DEFAULT 0,
  created_at DATETIME DEFAULT CURRENT_TIMESTAMP,
  UNIQUE(user_id, platform)
);

-- User interests/skills
CREATE TABLE user_interests (
  id TEXT PRIMARY KEY DEFAULT (uuid()),
  user_id TEXT NOT NULL REFERENCES users(id) ON DELETE CASCADE,
  category TEXT NOT NULL, -- 'language', 'domain', 'technology', 'interest'
  name TEXT NOT NULL,
  experience_level TEXT, -- 'beginner', 'intermediate', 'advanced', 'expert'
  years_experience INTEGER,
  display_order INTEGER DEFAULT 0,
  created_at DATETIME DEFAULT CURRENT_TIMESTAMP,
  UNIQUE(user_id, category, name)
);

CREATE INDEX idx_user_interests_category ON user_interests(category);
CREATE INDEX idx_user_interests_name ON user_interests(name);

-- Presence & Status
CREATE TABLE user_presence (
  user_id TEXT PRIMARY KEY REFERENCES users(id) ON DELETE CASCADE,

  custom_emoji TEXT,
  custom_text TEXT,
  custom_expires_at DATETIME,
  last_heartbeat_at DATETIME,
  updated_at DATETIME DEFAULT CURRENT_TIMESTAMP
);

  condition TEXT NOT NULL,
  target_status TEXT NOT NULL,
  custom_message TEXT,
  enabled BOOLEAN DEFAULT TRUE,
  priority INTEGER DEFAULT 0,
  created_at DATETIME DEFAULT CURRENT_TIMESTAMP
);

-- Friend requests
CREATE TABLE friend_requests (
  id TEXT PRIMARY KEY DEFAULT (uuid()),
  from_user_id TEXT NOT NULL REFERENCES users(id) ON DELETE CASCADE,
  to_user_id TEXT NOT NULL REFERENCES users(id) ON DELETE CASCADE,
  message TEXT, -- Optional message with request
  status TEXT NOT NULL DEFAULT 'pending', -- 'pending', 'accepted', 'declined', 'cancelled'
  created_at DATETIME DEFAULT CURRENT_TIMESTAMP,
  responded_at DATETIME,
  UNIQUE(from_user_id, to_user_id)

CREATE INDEX idx_friend_requests_from ON friend_requests(from_user_id, status);

CREATE TABLE friendships (

  friend_id TEXT NOT NULL REFERENCES users(id) ON DELETE CASCADE,

  favorite BOOLEAN DEFAULT FALSE,
  created_at DATETIME DEFAULT CURRENT_TIMESTAMP,
  UNIQUE(user_id, friend_id)
);

CREATE INDEX idx_friendships_friend ON friendships(friend_id);

  added_at DATETIME DEFAULT CURRENT_TIMESTAMP,
  PRIMARY KEY(group_id, friend_id)
);

-- Blocked users
CREATE TABLE blocked_users (
  user_id TEXT NOT NULL REFERENCES users(id) ON DELETE CASCADE,
  blocked_id TEXT NOT NULL REFERENCES users(id) ON DELETE CASCADE,

  created_at DATETIME DEFAULT CURRENT_TIMESTAMP,
  PRIMARY KEY(user_id, blocked_id)
```
