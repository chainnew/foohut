**FooHut Studio** is the AI-native creative coding playground where ideas become reality in seconds. It's where napkin sketches transform into interactive visualizations, where a single prompt spawns full applications, and where developers, designers, and creators collaborate in real-time on living documents that blend prose, code, and visual magic.

**The Vision:** *Imagine if Figma, CodeSandbox, Jupyter Notebooks, and Claude had a baby â€” that's FooHut Studio.*

## 3. Artifact System

### 3.1 What is an Artifact?

An **Artifact** is a self-contained, interactive content block that can contain code, visuals, data, or documents. Artifacts are the building blocks of Studio documents.

```typescript
interface Artifact {
  id: string;
  type: ArtifactType;
  title: string;
  content: ArtifactContent;
  position: { x: number; y: number };
  size: { width: number; height: number };
  connections: ArtifactConnection[];
  settings: ArtifactSettings;
  version: number;
  created_at: Date;
  updated_at: Date;
  created_by: string;
}

type ArtifactType = 
  | 'markdown'      // Rich text document
  | 'code'          // Code sandbox
  | 'notebook'      // Jupyter-style cells
  | 'canvas'        // Drawing/whiteboard
  | 'diagram'       // Flowchart/architecture
  | 'chart'         // Data visualization
  | 'three'         // 3D scene
  | 'webgl'         // Custom WebGL
  | 'html'          // HTML/CSS/JS preview
  | 'react'         // React component
  | 'api'           // API playground
  | 'database'      // SQL/data explorer
  | 'terminal'      // Shell/CLI
  | 'video'         // Video embed
  | 'image'         // Image with annotations
  | 'pdf'           // PDF viewer
  | 'embed'         // External embed
  | 'ai-chat'       // AI conversation
  | 'form'          // Interactive form
  | 'table'         // Spreadsheet-like
  | 'timeline'      // Timeline/gantt
  | 'mindmap'       // Mind map
  | 'presentation'  // Slide deck
  | 'audio'         // Audio player/visualizer
  | 'shader'        // GLSL shader
  | 'game'          // Game engine
  ;
```

## 9. Database Schema

```sql
-- =====================================================
-- STUDIO CORE TABLES
-- =====================================================

-- Projects (workspaces)
CREATE TABLE studio_projects (
  id TEXT PRIMARY KEY DEFAULT (uuid()),
  owner_id TEXT NOT NULL REFERENCES users(id) ON DELETE CASCADE,
  org_id TEXT REFERENCES organizations(id) ON DELETE CASCADE,
  name TEXT NOT NULL,
  slug TEXT NOT NULL,
  description TEXT,
  thumbnail_url TEXT,
  template_id TEXT,
  visibility TEXT DEFAULT 'private', -- 'private', 'unlisted', 'public'
  settings TEXT DEFAULT '{}', -- JSON
  created_at DATETIME DEFAULT CURRENT_TIMESTAMP,
  updated_at DATETIME DEFAULT CURRENT_TIMESTAMP,
  deleted_at DATETIME,
  UNIQUE(owner_id, slug)
);

CREATE INDEX idx_studio_projects_owner ON studio_projects(owner_id);
CREATE INDEX idx_studio_projects_visibility ON studio_projects(visibility);

-- Project collaborators
CREATE TABLE studio_project_collaborators (
  project_id TEXT NOT NULL REFERENCES studio_projects(id) ON DELETE CASCADE,
  user_id TEXT NOT NULL REFERENCES users(id) ON DELETE CASCADE,
  role TEXT NOT NULL DEFAULT 'viewer', -- 'owner', 'editor', 'commenter', 'viewer'
  invited_by TEXT REFERENCES users(id),
  joined_at DATETIME DEFAULT CURRENT_TIMESTAMP,
  PRIMARY KEY(project_id, user_id)
);

-- Artifacts
CREATE TABLE studio_artifacts (
  id TEXT PRIMARY KEY DEFAULT (uuid()),
  project_id TEXT NOT NULL REFERENCES studio_projects(id) ON DELETE CASCADE,
  type TEXT NOT NULL, -- 'markdown', 'code', 'three', 'chart', etc.
  title TEXT NOT NULL,
  content TEXT, -- JSON blob for artifact content
  position_x REAL DEFAULT 0,
  position_y REAL DEFAULT 0,
  width REAL DEFAULT 400,
  height REAL DEFAULT 300,
  z_index INTEGER DEFAULT 0,
  collapsed BOOLEAN DEFAULT FALSE,
  locked BOOLEAN DEFAULT FALSE,
  settings TEXT DEFAULT '{}', -- JSON for artifact-specific settings
  created_by TEXT REFERENCES users(id),
  created_at DATETIME DEFAULT CURRENT_TIMESTAMP,
  updated_at DATETIME DEFAULT CURRENT_TIMESTAMP
);

CREATE INDEX idx_studio_artifacts_project ON studio_artifacts(project_id);
CREATE INDEX idx_studio_artifacts_type ON studio_artifacts(type);

-- Artifact connections
CREATE TABLE studio_artifact_connections (
  id TEXT PRIMARY KEY DEFAULT (uuid()),
  project_id TEXT NOT NULL REFERENCES studio_projects(id) ON DELETE CASCADE,
  source_id TEXT NOT NULL REFERENCES studio_artifacts(id) ON DELETE CASCADE,
  target_id TEXT NOT NULL REFERENCES studio_artifacts(id) ON DELETE CASCADE,
  connection_type TEXT NOT NULL, -- 'data', 'trigger', 'embed', 'reference'
  mapping TEXT, -- JSON for field mappings
  settings TEXT DEFAULT '{}',
  created_at DATETIME DEFAULT CURRENT_TIMESTAMP
);

CREATE INDEX idx_studio_connections_source ON studio_artifact_connections(source_id);
CREATE INDEX idx_studio_connections_target ON studio_artifact_connections(target_id);

-- Artifact versions (history)
CREATE TABLE studio_artifact_versions (
  id TEXT PRIMARY KEY DEFAULT (uuid()),
  artifact_id TEXT NOT NULL REFERENCES studio_artifacts(id) ON DELETE CASCADE,
  version_number INTEGER NOT NULL,
  content TEXT NOT NULL,
  created_by TEXT REFERENCES users(id),
  created_at DATETIME DEFAULT CURRENT_TIMESTAMP,
  message TEXT,
  UNIQUE(artifact_id, version_number)
);

CREATE INDEX idx_studio_versions_artifact ON studio_artifact_versions(artifact_id);

-- Code sandbox files
CREATE TABLE studio_sandbox_files (
  id TEXT PRIMARY KEY DEFAULT (uuid()),
  artifact_id TEXT NOT NULL REFERENCES studio_artifacts(id) ON DELETE CASCADE,
  path TEXT NOT NULL, -- e.g., '/src/App.tsx'
  content TEXT,
  language TEXT,
  created_at DATETIME DEFAULT CURRENT_TIMESTAMP,
  updated_at DATETIME DEFAULT CURRENT_TIMESTAMP,
  UNIQUE(artifact_id, path)
);

CREATE INDEX idx_studio_files_artifact ON studio_sandbox_files(artifact_id);

-- Comments
CREATE TABLE studio_comments (
  id TEXT PRIMARY KEY DEFAULT (uuid()),
  project_id TEXT NOT NULL REFERENCES studio_projects(id) ON DELETE CASCADE,
  artifact_id TEXT REFERENCES studio_artifacts(id) ON DELETE CASCADE,
  parent_id TEXT REFERENCES studio_comments(id) ON DELETE CASCADE,
  user_id TEXT NOT NULL REFERENCES users(id),
  content TEXT NOT NULL,
  annotation_type TEXT DEFAULT 'comment', -- 'comment', 'bug', 'suggestion', 'question'
  position TEXT, -- JSON for line/character position
  resolved BOOLEAN DEFAULT FALSE,
  resolved_by TEXT REFERENCES users(id),
  resolved_at DATETIME,
  created_at DATETIME DEFAULT CURRENT_TIMESTAMP,
  updated_at DATETIME DEFAULT CURRENT_TIMESTAMP
);

CREATE INDEX idx_studio_comments_project ON studio_comments(project_id);
CREATE INDEX idx_studio_comments_artifact ON studio_comments(artifact_id);

-- AI conversations
CREATE TABLE studio_ai_conversations (
  id TEXT PRIMARY KEY DEFAULT (uuid()),
  project_id TEXT NOT NULL REFERENCES studio_projects(id) ON DELETE CASCADE,
  artifact_id TEXT REFERENCES studio_artifacts(id) ON DELETE CASCADE,
  user_id TEXT NOT NULL REFERENCES users(id),
  messages TEXT NOT NULL, -- JSON array of messages
  model TEXT DEFAULT 'claude-3.5-sonnet',
  tokens_used INTEGER DEFAULT 0,
  created_at DATETIME DEFAULT CURRENT_TIMESTAMP,
  updated_at DATETIME DEFAULT CURRENT_TIMESTAMP
);

-- Deployments
CREATE TABLE studio_deployments (
  id TEXT PRIMARY KEY DEFAULT (uuid()),
  project_id TEXT NOT NULL REFERENCES studio_projects(id) ON DELETE CASCADE,
  deployed_by TEXT NOT NULL REFERENCES users(id),
  provider TEXT NOT NULL, -- 'foohut', 'vercel', 'netlify', etc.
  url TEXT,
  status TEXT DEFAULT 'pending', -- 'pending', 'building', 'live', 'failed'
  build_logs TEXT,
  commit_hash TEXT,
  environment TEXT DEFAULT 'production',
  created_at DATETIME DEFAULT CURRENT_TIMESTAMP,
  completed_at DATETIME
);

CREATE INDEX idx_studio_deployments_project ON studio_deployments(project_id);

-- Templates
CREATE TABLE studio_templates (
  id TEXT PRIMARY KEY,
  name TEXT NOT NULL,
  description TEXT,
  category TEXT NOT NULL,
  thumbnail_url TEXT,
  content TEXT NOT NULL, -- JSON blob of template structure
  artifacts TEXT NOT NULL, -- JSON array of artifact types included
  author_id TEXT REFERENCES users(id),
  is_official BOOLEAN DEFAULT FALSE,
  usage_count INTEGER DEFAULT 0,
  created_at DATETIME DEFAULT CURRENT_TIMESTAMP
);

CREATE INDEX idx_studio_templates_category ON studio_templates(category);

-- Assets (images, fonts, etc.)
CREATE TABLE studio_assets (
  id TEXT PRIMARY KEY DEFAULT (uuid()),
  project_id TEXT NOT NULL REFERENCES studio_projects(id) ON DELETE CASCADE,
  name TEXT NOT NULL,
  type TEXT NOT NULL, -- 'image', 'font', 'video', 'audio', 'model'
  url TEXT NOT NULL,
  size_bytes INTEGER,
  mime_type TEXT,
  metadata TEXT, -- JSON for dimensions, duration, etc.
  uploaded_by TEXT REFERENCES users(id),
  created_at DATETIME DEFAULT CURRENT_TIMESTAMP
);

CREATE INDEX idx_studio_assets_project ON studio_assets(project_id);
CREATE INDEX idx_studio_assets_type ON studio_assets(type);

-- Real-time presence
CREATE TABLE studio_presence (
  project_id TEXT NOT NULL REFERENCES studio_projects(id) ON DELETE CASCADE,
  user_id TEXT NOT NULL REFERENCES users(id) ON DELETE CASCADE,
  cursor_x REAL,
  cursor_y REAL,
  artifact_id TEXT REFERENCES studio_artifacts(id),
  selection TEXT, -- JSON for text selection
  viewport TEXT, -- JSON for zoom/pan state
  last_seen_at DATETIME DEFAULT CURRENT_TIMESTAMP,
  PRIMARY KEY(project_id, user_id)
);

-- =====================================================
-- INDEXES FOR PERFORMANCE
-- =====================================================

CREATE INDEX idx_studio_projects_updated ON studio_projects(updated_at DESC);
CREATE INDEX idx_studio_artifacts_updated ON studio_artifacts(updated_at DESC);
CREATE INDEX idx_studio_presence_lastseen ON studio_presence(last_seen_at);
```
